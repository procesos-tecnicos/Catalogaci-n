function sleep(ms) {
    return new Promise(function (r) {
        setTimeout(r, ms);
    });
}

async function translateApi(text) {
    const url =
        "https://api.mymemory.translated.net/get?q=" +
        encodeURIComponent(text.slice(0, 450)) +
        "&langpair=en|es";
    const res = await fetch(url);
    if (!res.ok) throw new Error("HTTP " + res.status);
    const data = await res.json();
    if (data.responseStatus !== 200) throw new Error(data.responseDetails || "API error");
    return data.responseData.translatedText;
}

async function main() {
    const strings = JSON.parse(fs.readFileSync(STRINGS, "utf8"));
    const helpers = JSON.parse(fs.readFileSync(HELPERS, "utf8"));
    const reverse = {};
    Object.keys(helpers.exact || {}).forEach(function (es) {
        reverse[helpers.exact[es]] = es;
    });

    const glossary = fs.existsSync(OUT) ? JSON.parse(fs.readFileSync(OUT, "utf8")) : {};
    let added = 0;
    let skipped = 0;

    for (let i = 0; i < strings.length; i++) {
        const en = strings[i];
        if (glossary[en]) {
            skipped++;
            continue;
        }
        if (MANUAL[en]) {
            glossary[en] = MANUAL[en];
            added++;
            continue;
        }
        if (EN_TO_ES_FALLBACK[en]) {
            glossary[en] = EN_TO_ES_FALLBACK[en];
            added++;
            continue;
        }
        if (reverse[en]) {
            glossary[en] = reverse[en];
            added++;
            continue;
        }
        try {
            glossary[en] = await translateApi(en);
            added++;
            if (i % 20 === 0) console.log(i + 1 + "/" + strings.length, en.slice(0, 50));
            await sleep(350);
        } catch (e) {
            console.warn("FAIL:", en.slice(0, 60), e.message);
            glossary[en] = en;
        }
    }

    fs.writeFileSync(OUT, JSON.stringify(glossary, null, 2) + "\n", "utf8");
    console.log("Escrito:", OUT, "nuevas:", added, "omitidas:", skipped, "total:", Object.keys(glossary).length);
}

main().catch(function (e) {
    console.error(e);
    process.exit(1);
});
